\begin{problem}[Irregularities]

Consider the following language \( L \):
\[
  L = \{ x^i y^j z^k \mid k = i + j \}.
\]
\begin{enumerate}[label=(\alph*)]
  \item Prove that \( L \) is irregular with the pumping lemma.
  \item Prove that \( L \) is irregular using the Myhill-Nerode theorem.
\end{enumerate}

\end{problem}
